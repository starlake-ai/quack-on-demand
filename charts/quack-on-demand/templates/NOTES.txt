Quack on Demand has been installed.

# Verify the manager is up

  kubectl --namespace {{ .Release.Namespace }} wait --for=condition=Ready \
    pod -l app.kubernetes.io/instance={{ .Release.Name }} --timeout=120s

# Port-forward the admin UI / REST

  kubectl --namespace {{ .Release.Namespace }} port-forward \
    svc/{{ include "qod.fullname" . }} {{ .Values.service.rest.port }}:{{ .Values.service.rest.port }}

  Then open http://localhost:{{ .Values.service.rest.port }}/ui/
{{- if .Values.admin.password }}
  Login: {{ .Values.admin.username }} / (chart-managed admin password)
{{- else if .Values.admin.existingSecret }}
  Login: {{ .Values.admin.username }} / (value of {{ .Values.admin.existingSecret }}/{{ .Values.admin.existingSecretKey }})
{{- else }}
  WARNING: no admin password set. The manager will seed a default admin
  but you should rotate via SL_QUACK_ADMIN_PASSWORD or chart values
  before exposing the manager.
{{- end }}

# Port-forward FlightSQL (gRPC)

  kubectl --namespace {{ .Release.Namespace }} port-forward \
    svc/{{ include "qod.fullname" . }}-flightsql {{ .Values.service.flightsql.port }}:{{ .Values.service.flightsql.port }}

  JDBC URL template (replace <user>/<password>):

    jdbc:arrow-flight-sql://localhost:{{ .Values.service.flightsql.port }}?useEncryption={{ .Values.flightsql.tls.enabled }}{{- if .Values.flightsql.tls.enabled }}&disableCertificateVerification=true{{- end }}&user={{ .Values.bootstrap.tenant }}/{{ .Values.bootstrap.pool }}/<user>&password=<password>

# Watch the Quack node pods the manager spawns

  kubectl --namespace {{ .Release.Namespace }} get pods -l managed-by=quack-on-demand -w

- Replicas default to 1: multi-manager active-active is not yet safe.
  See RESILIENCE.md and issue #11.
- The manager calls the K8s API to create / delete pods + services in
  this namespace. RBAC scope is a Role, not a ClusterRole.
- {{- if eq .Values.metrics.sink "prometheus" }} /metrics is exposed on the rest-ui port. {{- else }} metrics sink: {{ .Values.metrics.sink }} (the /metrics endpoint is NOT exposed in cloud-push mode). {{- end }}
